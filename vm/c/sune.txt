void remove_from_waiting_queue(scheduler_t *scheduler, uint32_t jid) {
    dlist_iter_t iter;
    dlist_iter_init(&iter, &(*scheduler).waiting_queue);
    while(!dlist_iter_end(&iter)) {
        if (((job_t*)dlist_iter_current(&iter))->jid == jid) {
            dlist_iter_remove(&iter);
            break;
        } else {
            dlist_iter_next(&iter);
        }
    }
}
